💡 JavaScript란?

JavaScript는 웹 페이지에 동작을 추가하는 프로그래밍 언어입니다

HTML이 구조, CSS가 디자인이라면, JavaScript는 동작과 상호작용을 담당합니다.

🎯 JavaScript로 할 수 있는 것

  • 버튼 클릭, 폼 제출 등 사용자 이벤트 처리
  • 웹 페이지 내용을 동적으로 변경
  • 서버와 데이터 주고받기
  • 애니메이션과 시각 효과 구현
  • 브라우저에 데이터 저장

📦 변수 선언하기

변수란?

데이터를 저장하는 상자와 같습니다. 이름을 붙여서 나중에 다시 사용할 수 있습니다.

변수 선언 방법

// 1. let - 변경 가능한 변수 (추천)
let userName = "홍길동";
let age = 25;
userName = "김철수"; // 변경 가능 ✅

// 2. const - 상수 (변경 불가, 추천)
const PI = 3.14159;
const API_KEY = "abc123xyz";
// PI = 3.14; // ❌ 오류 발생!

// 3. var - 구식 방법 (사용 비권장)
var oldStyle = "구식 선언";
언제 무엇을 사용할까?
• const: 값이 변하지 않을 때 (기본으로 사용)
• let: 값이 변할 때만 사용
• var: 사용하지 마세요!

실용 예제

const siteName = "내 웹사이트"; // 사이트명은 변경 안됨
let loginCount = 0; // 로그인 횟수는 증가함
let isLoggedIn = false; // 로그인 상태는 변함
loginCount = loginCount + 1; // 1 증가
isLoggedIn = true; // 로그인 완료

📚 배열 (Array)

배열은 여러 값을 순서대로 저장하는 리스트입니다

// 배열 선언
let fruits = ["사과", "바나나", "오렌지"];
let numbers = [1, 2, 3, 4, 5];
let mixed = ["문자", 123, true]; // 다른 타입도 가능

// 배열 요소 접근 (인덱스는 0부터 시작!)
console.log(fruits[0]); // "사과"
console.log(fruits[1]); // "바나나"
console.log(fruits[2]); // "오렌지"

// 배열 길이
console.log(fruits.length); // 3

// 배열 수정
fruits[0] = "딸기";
console.log(fruits); // ["딸기", "바나나", "오렌지"]

// 배열에 요소 추가
fruits.push("포도"); // 끝에 추가
fruits.unshift("수박"); // 앞에 추가

// 배열에서 요소 제거
fruits.pop(); // 마지막 제거
fruits.shift(); // 첫번째 제거

🎁 객체 (Object)

객체는 관련된 데이터를 이름(키)과 값(value)의 쌍으로 저장합니다

// 객체 선언
let person = {
  name: "김철수",
  age: 30,
  city: "서울",
  isStudent: false
};

// 객체 속성 접근 (점 표기법)
console.log(person.name); // "김철수"
console.log(person.age); // 30

// 객체 속성 접근 (대괄호 표기법)
console.log(person["city"]); // "서울"

// 객체 속성 수정
person.age = 31;
person.city = "부산";

// 새 속성 추가
person.job = "개발자";
person.email = "kim@email.com";

// 속성 삭제
delete person.isStudent;

⚙️ 함수 (Functions)

함수는 재사용 가능한 코드 블록입니다

특정 작업을 수행하는 코드를 묶어서 이름을 붙이고, 필요할 때마다 호출할 수 있습니다.

1️⃣ 함수 선언문

// 함수 정의
function greet(name) {
  return "안녕하세요, " + name + "님!";
}

// 함수 호출
let message = greet("철수");
console.log(message); // "안녕하세요, 철수님!"

// 매개변수가 여러 개인 함수
function add(a, b) {
  return a + b;
}

let sum = add(5, 3);
console.log(sum); // 8

2️⃣ 함수 표현식

// 변수에 함수를 할당
const multiply = function(a, b) {
  return a * b;
};

let result = multiply(4, 7);
console.log(result); // 28

3️⃣ 화살표 함수 (ES6, 최신 방식)

// 기본 형태
const subtract = (a, b) => {
  return a - b;
};

// 축약형 (한 줄일 때 return 생략 가능)
const square = x => x * x;
const double = x => x * 2;

console.log(square(5)); // 25
console.log(double(10)); // 20

⚙️ 함수 실전 예제

예제 1: 계산기 함수

function calculator(num1, num2, operator) {
  if (operator === "+") {
    return num1 + num2;
  } else if (operator === "-") {
    return num1 - num2;
  } else if (operator === "*") {
    return num1 * num2;
  } else if (operator === "/") {
    return num1 / num2;
  }
}

console.log(calculator(10, 5, "+")); // 15
console.log(calculator(10, 5, "*")); // 50

예제 2: 할인 가격 계산

const calculateDiscount = (price, discountRate) => {
  const discount = price * (discountRate / 100);
  const finalPrice = price - discount;
  return finalPrice;
};

let originalPrice = 50000;
let salePrice = calculateDiscount(originalPrice, 20);
console.log(`할인가: ${salePrice}원`); // 할인가: 40000원

🔀 조건문 (if-else)

조건에 따라 다른 코드를 실행합니다

// 기본 if-else
let age = 20;

if (age >= 18) {
  console.log("성인입니다");
} else {
  console.log("미성년자입니다");
}

// 다중 조건 (else if)
let score = 85;

if (score >= 90) {
  console.log("A등급입니다!");
} else if (score >= 80) {
  console.log("B등급입니다!");
} else if (score >= 70) {
  console.log("C등급입니다!");
} else {
  console.log("더 노력하세요!");
}

비교 연산자

// 같다 / 다르다
5 == 5 // true (값과 타입이 같다)
5 != 3 // true (같지 않다)

// 크다 / 작다
10 > 5 // true
10 < 5 // false
10 >= 10 // true
10 <= 5 // false

// 논리 연산자
age >= 18 && age <= 65 // AND (그리고)
age < 18 || age > 65 // OR (또는)
!(age < 18) // NOT (부정)

삼항 연산자 (간단한 조건)

// 조건 ? 참일 때 : 거짓일 때
let result = score >= 80 ? "합격" : "불합격";
let message = age >= 18 ? "입장 가능" : "입장 불가";

🔁 반복문 - for 문

같은 작업을 여러 번 반복할 때 사용합니다

기본 for 반복문

// 0부터 4까지 출력
for (let i = 0; i < 5; i++) {
  console.log("숫자:", i);
}
// 출력: 0, 1, 2, 3, 4

// 1부터 10까지의 합계
let sum = 0;
for (let i = 1; i <= 10; i++) {
  sum += i; // sum = sum + i
}
console.log("합계:", sum); // 55

배열 반복 - for 문

let fruits = ["사과", "바나나", "오렌지"];

for (let i = 0; i < fruits.length; i++) {
  console.log(i + ": " + fruits[i]);
}
// 0: 사과 // 1: 바나나 // 2: 오렌지

배열 반복 - for...of (더 간단)

let colors = ["빨강", "파랑", "초록"];

for (let color of colors) {
  console.log("색상:", color);
}
// 색상: 빨강 // 색상: 파랑 // 색상: 초록

🔁 반복문 - forEach와 객체

배열 반복 - forEach 메서드

let numbers = [1, 2, 3, 4, 5];

// forEach: 각 요소마다 함수 실행
numbers.forEach(function(number, index) {
  console.log(`${index}번째: ${number}`);
});

// 화살표 함수로도 사용 가능
numbers.forEach((num) => {
  console.log(num * 2); // 각 숫자를 2배로
});

객체 반복 - for...in

let student = {
  name: "김학생",
  age: 20,
  grade: "A",
  school: "한국대학교"
};

// 객체의 모든 속성 순회
for (let key in student) {
  console.log(`${key}: ${student[key]}`);
}
// name: 김학생 // age: 20 // grade: A // school: 한국대학교

  sum += score;

🌳 DOM이란 무엇인가?

DOM (Document Object Model)

HTML 문서를 JavaScript로 조작할 수 있게 해주는 인터페이스

🔑 핵심 개념

  • HTML의 모든 요소는 객체(Object)로 표현됩니다
  • JavaScript를 통해 요소를 찾고, 수정하고, 생성하고, 삭제할 수 있습니다
  • 웹 페이지를 실시간으로 변경할 수 있습니다

🏗️ 비유로 이해하기

HTML 문서는 나무(Tree) 구조입니다:

  • <html> = 뿌리
  • <head>, <body> = 큰 가지
  • <div>, <p>, <span> = 작은 가지와 잎
💡 왜 DOM을 배워야 할까?
웹 페이지에 사용자 상호작용을 추가하려면 DOM 조작이 필수입니다!
예: 버튼 클릭 시 내용 변경, 장바구니 추가, 댓글 작성 등

🎯 요소 선택하기 - 기초

1️⃣ 단일 요소 선택

// ID로 선택
const title = document.getElementById("main-title");

// CSS 선택자로 선택
const firstButton = document.querySelector(".btn");
const navElement = document.querySelector("nav");
const mainTitle = document.querySelector("#main-title");

2️⃣ 여러 요소 선택

// 클래스로 여러 개 선택
const allButtons = document.querySelectorAll(".btn");

// 태그로 여러 개 선택
const listItems = document.querySelectorAll("ul li");

// 속성 선택자 사용
const textInputs = document.querySelectorAll("input[type='text']");
⭐ querySelector vs querySelectorAll
• querySelector: 첫 번째 요소 1개만 반환
• querySelectorAll: 일치하는 모든 요소를 NodeList로 반환
💡 CSS 선택자를 그대로 사용할 수 있습니다!
".container .item", "div > p", "button:hover" 등

🎯 요소 선택하기 - 심화

부모-자식 관계 탐색

const container = document.querySelector(".container");

// 자식 요소들
const children = container.children; // 모든 자식 요소
const firstChild = container.firstElementChild; // 첫 번째 자식
const lastChild = container.lastElementChild; // 마지막 자식

// 부모 요소
const parent = container.parentElement;

형제 요소 탐색

const currentElement = document.querySelector("#current");

// 다음 형제
const nextSibling = currentElement.nextElementSibling;

// 이전 형제
const prevSibling = currentElement.previousElementSibling;

실용 예제: 가장 가까운 조상 요소 찾기

const deleteBtn = document.querySelector(".delete-btn");

// 버튼의 가장 가까운 .card 부모 찾기
const card = deleteBtn.closest(".card");
💡 closest() 메서드는 현재 요소부터 시작해서 조건에 맞는 가장 가까운 조상 요소를 찾습니다. 이벤트 위임에서 매우 유용합니다!

📝 요소 내용 조작하기

• textContent: 단순 텍스트만 다룰 때
• innerHTML: HTML 태그를 포함할 때

데이터 속성 (data-*) 활용

// HTML: <div data-user-id="123" data-user-name="김철수"></div>

🎯 클래스 조작하기 (중요!)

클래스 조작은 DOM 조작에서 가장 많이 사용되는 기술입니다!

실용 예제: 탭 메뉴 전환

const tabs = document.querySelectorAll(".tab");

function switchTab(clickedTab) {
  // 모든 탭에서 active 클래스 제거
  tabs.forEach(tab => tab.classList.remove("active"));

  // 클릭된 탭에만 active 클래스 추가
  clickedTab.classList.add("active");
}

// 각 탭에 클릭 이벤트 등록
tabs.forEach(tab => {
  tab.addEventListener("click", () => switchTab(tab));
});

// 요소 선택
const todoInput = document.querySelector("#todoInput");
const addBtn = document.querySelector("#addBtn");
const todoList = document.querySelector("#todoList");

// 할일 추가 함수
function addTodo() {
  const text = todoInput.value.trim();

  if (text === "") {
    alert("할일을 입력하세요!");
    return;
  }

  // 새 li 요소 생성
  const li = document.createElement("li");
  li.className = "todo-item";

  // 텍스트 span 생성
  const span = document.createElement("span");
  span.textContent = text;
  span.addEventListener("click", function() {
    li.classList.toggle("completed");
  });

  // 삭제 버튼 생성
  const deleteBtn = document.createElement("button");
  deleteBtn.textContent = "삭제";
  deleteBtn.addEventListener("click", function() {
    li.remove();
  });

  // li에 추가
  li.appendChild(span);
  li.appendChild(deleteBtn);

  // 목록에 추가
  todoList.appendChild(li);

  // 입력칸 비우기
  todoInput.value = "";
  todoInput.focus();
}

// 버튼 클릭 시
addBtn.addEventListener("click", addTodo);

// Enter 키 입력 시
todoInput.addEventListener("keypress", function(e) {
  if (e.key === "Enter") {
    addTodo();
  }
});

📗 2회차: 이벤트 처리와 폼 검증

사용자 상호작용 완벽 마스터

🎬 이벤트란?

이벤트 (Event)

웹 페이지에서 발생하는 모든 사건

클릭, 입력, 스크롤, 마우스 움직임 등

대표적인 이벤트들

  • 🖱️ 마우스 이벤트: click, dblclick, mouseover, mouseout
  • ⌨️ 키보드 이벤트: keydown, keyup, keypress
  • 📝 폼 이벤트: submit, focus, blur, change, input
  • 🌐 페이지 이벤트: load, resize, scroll
이벤트를 잘 다루면 사용자 경험(UX)이 획기적으로 좋아집니다!

👂 이벤트 리스너 등록하기

기본 방법: addEventListener

const button = document.querySelector("#myButton");

// 방법 1: 일반 함수
button.addEventListener("click", function(event) {
  console.log("버튼이 클릭되었습니다!");
  console.log(event); // 이벤트 객체
});

// 방법 2: 화살표 함수 (추천)
button.addEventListener("click", (e) => {
  console.log("화살표 함수로 처리!");
});

// 방법 3: 함수 참조
function handleClick(event) {
  console.log("함수 참조로 처리!");
}

button.addEventListener("click", handleClick);

이벤트 리스너 제거

// 제거하려면 같은 함수를 참조해야 함
button.removeEventListener("click", handleClick);
⚠️ 주의: 익명 함수로 등록한 이벤트는 제거할 수 없습니다! 제거가 필요하면 함수를 변수에 저장하세요.

📝 폼 이벤트

const input = document.querySelector("#username");
const form = document.querySelector("#myForm");

// focus: 입력칸을 선택했을 때
input.addEventListener("focus", (e) => {
  e.target.style.backgroundColor = "#e3f2fd";
  e.target.style.borderColor = "blue";
});

// blur: 입력칸에서 벗어났을 때
input.addEventListener("blur", (e) => {
  e.target.style.backgroundColor = "";
  e.target.style.borderColor = "";
});

// input: 입력값이 변할 때마다 (실시간)
input.addEventListener("input", (e) => {
  console.log(`현재 값: ${e.target.value}`);

  // 글자 수 표시
  const counter = document.querySelector(".char-counter");
  counter.textContent = `${e.target.value.length}자`;
});

// change: 값이 변경되고 포커스를 잃었을 때
input.addEventListener("change", (e) => {
  console.log(`최종 값: ${e.target.value}`);
});

// submit: 폼 제출 시
form.addEventListener("submit", (e) => {
  e.preventDefault(); // 페이지 새로고침 막기!

  const username = input.value.trim();

  if (username === "") {
    alert("이름을 입력하세요!");
    return;
  }

  console.log("폼 제출:", username);
});
⚠️ 중요: 폼 submit 이벤트에서는 반드시 preventDefault()를 호출해야 페이지가 새로고침되지 않습니다!

📦 이벤트 객체 (Event Object)

이벤트 객체란?

이벤트가 발생할 때 자동으로 생성되는 객체로, 이벤트에 대한 모든 정보를 담고 있습니다.

button.addEventListener("click", function(event) {
  // 기본 정보
  console.log("이벤트 타입:", event.type); // "click"
  console.log("이벤트 대상:", event.target); // 클릭된 요소
  console.log("현재 대상:", event.currentTarget); // 리스너가 있는 요소

  // 마우스 위치 정보
  console.log("화면 기준:", event.clientX, event.clientY);
  console.log("페이지 기준:", event.pageX, event.pageY);
  console.log("요소 기준:", event.offsetX, event.offsetY);

  // 수정자 키 상태
  console.log("Ctrl 키:", event.ctrlKey);
  console.log("Shift 키:", event.shiftKey);
  console.log("Alt 키:", event.altKey);

  // 이벤트 제어
  event.preventDefault(); // 기본 동작 막기
  event.stopPropagation(); // 이벤트 버블링 막기
});
target vs currentTarget
• target: 실제로 클릭된 요소
• currentTarget: 이벤트 리스너가 등록된 요소

🎯 이벤트 위임 (Event Delegation)

이벤트 위임이란?

부모 요소에 이벤트 리스너를 등록하여 자식 요소들의 이벤트를 한 번에 처리하는 기법

❌ 비효율적인 방법

// 모든 버튼에 개별로 이벤트 등록 (나쁨)
const buttons = document.querySelectorAll(".btn");
buttons.forEach(btn => {
  btn.addEventListener("click", handleClick);
});

// 문제: 나중에 추가된 버튼에는 이벤트가 없음!

✅ 효율적인 방법 - 이벤트 위임

// 부모 요소에 한 번만 등록 (좋음)
const container = document.querySelector(".container");

container.addEventListener("click", function(event) {
  // 클릭된 요소가 버튼인지 확인
  if (event.target.classList.contains("btn")) {
    console.log("버튼 클릭:", event.target.textContent);

    // 버튼 타입별 처리
    if (event.target.classList.contains("delete-btn")) {
      deleteItem(event.target);
    } else if (event.target.classList.contains("edit-btn")) {
      editItem(event.target);
    }
  }
});

  }
}

}

const loginForm = document.querySelector("#loginForm");

loginForm.addEventListener("submit", function(event) {
  event.preventDefault(); // 제출 막기

  // 입력값 가져오기 (공백 제거)
  const username = usernameInput.value.trim();
  const password = passwordInput.value.trim();

  // 기본 검증
  if (username === "") {
    alert("사용자명을 입력해주세요!");
    usernameInput.focus(); // 포커스 이동
    return;
  }

  if (password === "") {
    alert("비밀번호를 입력해주세요!");
    passwordInput.focus();
    return;
  }

  if (password.length < 6) {
    alert("비밀번호는 6자 이상이어야 합니다!");
    passwordInput.focus();
    return;
  }

  // 검증 통과!
  console.log("로그인 시도:", { username, password });
  alert("로그인 성공!");
});

🔍 정규표현식으로 고급 검증

// 검증 패턴 정의
const validators = {
  email: {
    pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
    message: "올바른 이메일 주소를 입력해주세요"
  },
  password: {
    pattern: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{8,}$/,
    message: "비밀번호는 8자 이상, 대소문자, 숫자, 특수문자 포함"
  },
  phone: {
    pattern: /^01[016789]-?\d{3,4}-?\d{4}$/,
    message: "올바른 휴대폰 번호를 입력해주세요"
  }
};

// 검증 함수
function validateField(input, validatorKey) {
  const value = input.value.trim();
  const validator = validators[validatorKey];
  const feedback = document.querySelector(`#${input.id}-feedback`);

  if (value === "") {
    showError(input, feedback, "필수 입력 항목입니다");
    return false;
  }

  if (validator && !validator.pattern.test(value)) {
    showError(input, feedback, validator.message);
    return false;
  }

  showSuccess(input, feedback, "✓ 올바른 입력입니다");
  return true;
}

function showError(input, feedback, message) {
  input.classList.remove("valid");
  input.classList.add("invalid");
  feedback.textContent = message;
  feedback.className = "feedback error";
}

function showSuccess(input, feedback, message) {
  input.classList.remove("invalid");
  input.classList.add("valid");
  feedback.textContent = message;
  feedback.className = "feedback success";
}

// 사용
const emailInput = document.querySelector("#email");
emailInput.addEventListener("blur", function() {
  validateField(this, "email");
});

📱 전화번호 자동 포맷팅

const phoneInput = document.querySelector("#phone");

phoneInput.addEventListener("input", function() {
  // 숫자만 추출
  let value = this.value.replace(/[^0-9]/g, "");

  // 자동 포맷팅: 010-1234-5678
  if (value.length >= 3 && value.length <= 7) {
    value = value.replace(/(\d{3})(\d{1,4})/, "$1-$2");
  } else if (value.length >= 8) {
    value = value.replace(/(\d{3})(\d{4})(\d{0,4})/, "$1-$2-$3");
  }

  this.value = value;

  // 검증도 함께
  validateField(this, "phone");
});

  • 입력: 01012345678
  • 출력: 010-1234-5678
